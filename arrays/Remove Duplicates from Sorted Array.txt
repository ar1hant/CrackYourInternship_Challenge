class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int i = 0, j = 0, n = nums.size();
        int count = 0;
        while(j < n){
            while(j < n && nums[i] == nums[j])
                j++;
            count++;
            if(j >= n)
                break;
            nums[++i] = nums[j];
        }
        return count;
    }
};
